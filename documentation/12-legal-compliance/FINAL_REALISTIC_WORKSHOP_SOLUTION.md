```typescript
function getCurrentMechanicMode(mechanicId: string): 'workshop' | 'independent' {
  const mechanic = getMechanic(mechanicId)

  if (!mechanic.has_dual_mode) {
    return 'independent' // Pure independent mechanic
  }

  // Check if currently during workshop shift
  const now = new Date()
  const currentDay = now.toLocaleDateString('en-US', { weekday: 'lowercase' })
  const currentTime = now.toTimeString().substring(0, 5) // "14:30"

  if (mechanic.workshop_days.includes(currentDay)) {
    if (currentTime >= mechanic.workshop_shift_start &&
        currentTime < mechanic.workshop_shift_end) {
      return 'workshop' // During shift
    }
  }

  return 'independent' // Outside shift hours
}
```
